## Решения на задачите за стрингове

```c
#include <string.h>
#include <stdio.h>

int strcmp2(char[], char[]);
int strlen2(char[]);

int main() {
    char str[6] = {'h', 'i', '\0', 'h', 'i', '\0'};
    printf("%s\n", str); // hi
    printf("%zu - %d\n", strlen(str), strlen2(str)); // 2

    printf("%d - %d\n", strcmp("asd", "xyz"), strcmp2("asd", "xyz"));

    printf("%c - %d\n", 'a', 'a');
    printf("%c - %d\n", 'x', 'x');
    printf("%d\n", 'a' - 'x');

    for (int i = 'A'; i <= 'Z'; i++) {
        printf("%c", i);
    }
    printf("\n");
    for (int i = 65; i <= 90; i++) {
        printf("%c", i);
    }

    return 0;
}

int strcmp2(char str1[], char str2[]) {
    for (int i = 0; i < strlen2(str1); i++) {
        /* v1
        if (str1[i] < str2[i]) {
            return -1;
        }

        if (str1[i] > str2[i]) {
            return 1;
        }*/

        if (str1[i] != str2[i]) {
            return str1[i] - str2[i];
        }
    }

    return 0;
}

int strlen2(char str[]) {
    /* v1
    int i = 0;
    while(str[i] != '\0') {
        i++;
    }*/
    int i;

    for (i = 0; str[i] != '\0'; i++) {}

    return i;
}
```
